
## To delete => remote branch
 git push origin :newbranch
## To delete a local branch
git branch -d newbranch
git branch -D newbranch
## to list tags 
git tag
## delete tag 
## delete remote tag 
git push origin --delete v1.4
git push origin :refs/tags/v1.4
## delete local tag 
git tag -d v1.4
